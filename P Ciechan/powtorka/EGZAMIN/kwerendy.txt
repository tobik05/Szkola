2. SELECT avg(koszt) as "Średni koszt", SUM(koszt) as "Całkowity koszt", COUNT(id) as "Liczba uczestników" FROM uczestnicy
3. SELECT nazwa, nazwisko, email from uczestnicy inner join wyklady on uczestnicy.wyklady_id = wyklady.id where zaplacono >0
